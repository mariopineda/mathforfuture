Simplify:

  1. $ z^2 \times z^7$
  2. $\left( a^3 b^4\right) \left( a^8 b^3\right)$
  3. $\left(-2xy\right) \left(-5x^7y^3\right)$
  4. $\fraction{a^6}{a^3}$
  5. $\fraction{-12x^6y^5}{4xy^3}$

{% if true %}
  1. \[ \begin{align} & \left( \frac{m^5}{m} \right) \left( \frac{m^7}{m^9} \right) \\ = & m^{5-1}m^{7-9} \mbox{ (Quotient Law)}\\ = & m^4m^{-2} \\ = & m^{4+(-2)} \mbox{ (Product Law)}\\ = & \boxed{m^2} \end{align} \]
  2. \[ \begin{align} & \left( \frac{10a^7}{b^5} \right) \left( \frac{b^9}{2a^6} \right) \\ = & \frac{10a^7b^9}{2a^6b^5} \\ = & 5a^{7-6}b^{9-5} \mbox{ (Quotient Law)}\\ = & \boxed{5ab^4} \end{align} \]
  3. \[ \begin{align} & \left( \frac{f}{f^{-3}} \right) \left( \frac{f^{-2}}{f^5} \right) \\ = & f^{1-(-3)} f^{-2-5} \mbox{ (Quotient Law)}\\ = & f^{1+3} f^{-7}\\ = & f^4f^{-7} \\ = & f^{4+(-7)} \\ = & f^{-3} \\ = & \boxed{\frac{1}{f^3}} \mbox{ (Negative Exponent Law)} \end{align} \]
  4. \[ \begin{align} & \left( \frac{q^7r^{-1}}{s^{-2}t^5} \right)^{-3} \\ = & \frac{q^{-3(7)}r^{-1(-3)}}{s^{-2(-3)}t^{5(-3)}} \mbox{ (Power of a Power/Quotient Law)}\\ = & \frac{q^{-21}r^3}{s^6t^{-15}} \\ = & \boxed{\frac{t^{15}r^3}{s^6q^{21}}} \mbox{ (Negative Exponent Law)} \\ \end{align} \]
{% endif %}
